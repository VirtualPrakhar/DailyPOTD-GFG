=========================Java Solution==================================
class Solution {

    private long atMostK(String s, int k) {
        int left = 0;
        long count = 0;
        int distinct = 0;
        int[] freq = new int[26];

        for (int right = 0; right < s.length(); right++) {
            if (freq[s.charAt(right) - 'a'] == 0)
                distinct++;
            freq[s.charAt(right) - 'a']++;

            while (distinct > k) {
                freq[s.charAt(left) - 'a']--;
                if (freq[s.charAt(left) - 'a'] == 0)
                    distinct--;
                left++;
            }

            count += (right - left + 1);
        }
        return count;
    }

    public int countSubstr(String s, int k) {
        if (k == 0)
            return 0;
        return (int) (atMostK(s, k) - atMostK(s, k - 1));
    }
}

class Solution:
    def countSubstr(self, s: str, k: int) -> int:
        
        def atMostK(k):
            left = 0
            freq = {}
            distinct = 0
            count = 0

            for right in range(len(s)):
                if s[right] not in freq:
                    freq[s[right]] = 0
                    distinct += 1
                freq[s[right]] += 1

                while distinct > k:
                    freq[s[left]] -= 1
                    if freq[s[left]] == 0:
                        del freq[s[left]]
                        distinct -= 1
                    left += 1

                count += right - left + 1

            return count

        if k == 0:
            return 0
        return atMostK(k) - atMostK(k - 1)

class Solution {
  public:
    long long atMostK(string s, int k) {
        unordered_map<char, int> freq;
        int left = 0;
        long long count = 0;

        for (int right = 0; right < s.size(); right++) {
            freq[s[right]]++;

            while (freq.size() > k) {
                freq[s[left]]--;
                if (freq[s[left]] == 0)
                    freq.erase(s[left]);
                left++;
            }

            count += (right - left + 1);
        }
        return count;
    }

    int countSubstr(string s, int k) {
        if (k == 0)
            return 0;
        return atMostK(s, k) - atMostK(s, k - 1);
    }
};

class Solution {
    countSubstr(s, k) {

        const atMostK = (k) => {
            let left = 0;
            let count = 0;
            const freq = new Map();

            for (let right = 0; right < s.length; right++) {
                freq.set(s[right], (freq.get(s[right]) || 0) + 1);

                while (freq.size > k) {
                    freq.set(s[left], freq.get(s[left]) - 1);
                    if (freq.get(s[left]) === 0) {
                        freq.delete(s[left]);
                    }
                    left++;
                }

                count += (right - left + 1);
            }

            return count;
        };

        if (k === 0) return 0;
        return atMostK(k) - atMostK(k - 1);
    }
}
