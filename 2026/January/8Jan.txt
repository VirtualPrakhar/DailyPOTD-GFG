=========================Java Solution==================================
class Solution {
    public int countSubarrays(int[] arr, int k) {
        HashMap<Integer, Integer> freq = new HashMap<>();
        freq.put(0, 1); // base case

        int oddCount = 0;
        int result = 0;

        for (int num : arr) {
            if (num % 2 != 0) {
                oddCount++;
            }

            result += freq.getOrDefault(oddCount - k, 0);
            freq.put(oddCount, freq.getOrDefault(oddCount, 0) + 1);
        }

        return result;
    }
}

class Solution:
    def countSubarrays(self, arr, k):
        from collections import defaultdict

        freq = defaultdict(int)
        freq[0] = 1

        odd_count = 0
        result = 0

        for num in arr:
            if num % 2 != 0:
                odd_count += 1

            result += freq[odd_count - k]
            freq[odd_count] += 1

        return result

class Solution {
public:
    int countSubarrays(vector<int>& arr, int k) {
        unordered_map<int, int> freq;
        freq[0] = 1;

        int oddCount = 0;
        int result = 0;

        for (int num : arr) {
            if (num % 2 != 0) {
                oddCount++;
            }

            if (freq.find(oddCount - k) != freq.end()) {
                result += freq[oddCount - k];
            }

            freq[oddCount]++;
        }

        return result;
    }
};

        freq.set(0, 1);

        let oddCount = 0;
        let result = 0;

        for (const num of arr) {
            if (num % 2 !== 0) {
                oddCount++;
            }

            freq.set(oddCount, (freq.get(oddCount) || 0) + 1);
        }
